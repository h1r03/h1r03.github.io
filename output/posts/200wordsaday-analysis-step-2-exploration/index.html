    ps = soup.find_all('p', {'class': 'text-content'})
time_deltas = []
nwords = []
for p in ps:
    for p_date in p.next_siblings:
        if str(p_date).startswith('<p>published'):
            delta, nword= p_date.text.replace('published ', '').split('|')
            time_deltas.append(delta.strip())
            nwords.append(nword)
